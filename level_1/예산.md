```javascript
function solution(d, budget) {
  let sum = 0;
  const answer = d
    .sort((a, b) => a - b)
    .filter((list, idx) => {
      sum += list;
      if (sum <= budget) {
        return list;
      }
    });
  return answer.length;
}
```

function solution(d, budget) {
  return d
    .sort((a, b) => a - b)
    .reduce((count, price) => {
      return count + ((budget -= price) >= 0);
    }, 0);
}
