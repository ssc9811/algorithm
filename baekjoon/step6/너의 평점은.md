```javascript
const [[n], ...lists] = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n")
  .map((v) => v.split(" ").map((v) => +v));

lists.forEach((list) => {
  const [t, ...scores] = list;
  let sum = scores.reduce((acc, cur) => (acc += cur), 0);
  console.log(
    ((scores.filter((score) => score > sum / t).length / t) * 100).toFixed(3) +
      "%"
  );
});
```
